Attribute Groups
•Define groups of attributes with xs:attributeGroup
•Similar to xs:group
•Define at top level of schema with name attribute
•Refer to in type definitions or other attribute groups with ref attribute
<xs:attributeGroup name="common">
 <xs:attribute name="id" type="xs:ID" />
 <xs:attribute name="class" type="xs:NMTOKENS" />
</xs:attributeGroup>
<xs:element name="p">
 <xs:complexType mixed="true">
   <xs:group ref="html:inline" />
   <xs:attributeGroup ref="html:common" />
 </xs:complexType>
</xs:element>
