Extending Complex Types
<xs:complexType name="foreignAddress">
  <xs:complexContent>
    <xs:extension base="address">
      <xs:sequence>
        <xs:element ref="country" />
      </xs:sequence>
      <xs:attribute name="intCode"
                    type="xs:integer" />
    </xs:extension>
  </xs:complexContent>
</xs:complexType>
