XML Schema: Data Types
28
Length
•Units of length depend on data type
•Strings and URIs use characters
•Binary data types use bytes
•List types use number of items in list
•
<xs:simpleType name="dateRange">
 <xs:restriction>
  <xs:simpleType>
    <xs:list itemType="xs:date" />
  </xs:simpleType>
  <xs:minLength value="1" />
  <xs:maxLength value="2" />
 </xs:restriction>
</xs:simpleType>
<xs:simpleType name="airportCode">
 <xs:restriction base="xs:token">
  <xs:length value="3" />
 </xs:restriction>
</xs:simpleType>
